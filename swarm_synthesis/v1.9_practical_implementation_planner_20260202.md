# vajra_analysis_server.py
from transformer_lens import HookedTransformer
from mcp import Server, types
import torch

class VajraAnalysisServer:
    def __init__(self):
        # Use lightweight model for prototype
        self.model = HookedTransformer.from_pretrained("gpt2-small")
        self.sae_cache = {}
    
    async def analyze_activations(self, text: str) -> Dict[str, Any]:
        """Analyze model activations for interpretability"""
        with torch.no_grad():
            logits, cache = self.model.run_with_cache(text)
            
            # Extract key metrics
            attention_patterns = cache["pattern", -1]  # Final layer attention
            residual_stream = cache["resid_post", -1]  # Final residuals
            
            return {
                "attention_entropy": self._compute_attention_entropy(attention_patterns),
                "residual_norm": residual_stream.norm().item(),
                "interpretability_score": self._compute_interpretability(cache)
            }
    
    def _compute_attention_entropy(self, patterns):
        """Simple attention entropy calculation"""
        entropy = -torch.sum(patterns * torch.log(patterns + 1e-8), dim=-1)
        return entropy.mean().item()
